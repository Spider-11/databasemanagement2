RIGHT JOIN: THE ONLY SEE THOSE COLUMNS IN WHICH THE DATA EXISTS ON THE SECOND TABLE.

mysql> select student.name,student.section,student_marks.total_marks
        -> from student right join student_marks on student.rollno=student_marks.rollno;
