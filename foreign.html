THE FOREIGN KEY IDENTIFIES ANY COLUMN REFERENCING THE PRIMARY KEY IN ANOTHER TABLE.

mysql> create table marks(rollno int, section char(1),
        -> total_marks int,foreign key(rollno) references student(rollno));
    Query OK, 0 rows affected (0.68 sec)
