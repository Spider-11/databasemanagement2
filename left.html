LEFT JOIN: THE ONLY SEE THOSE COLUMNS IN WHICH THE DATA EXISTS ON THE FIRST TABLE.

mysql> select student.name,student.section,student_marks.total_marks
        -> from student left join student_marks on student.rollno=student_marks.rollno;
