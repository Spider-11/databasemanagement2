A NAMED PROCEDURE IS CREATED IN UNNAMED PROCEDURE SO THAT WE CAN IN THE UNNAMED PROCEDURE WHENEVER WE WANT.

declare  
        c number; 
    procedure addition(a number,b in out number) is 
    begin 
    dbms_output.put_line(a||' '||b); 
    b:=a+b; 
    end; 
    begin 
        c:=10; 
    addition(5,c); 
    dbms_output.put_line(c); 
    end; 

    Statement processed.
    5 10
    15
