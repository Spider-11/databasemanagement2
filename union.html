THE UNION OPERATOR JOINS THE OUTPUTS OF TWO OR MORE QUEIRIES INTO A SINGLE SET OF ROWS AND COLUMNS HAVING DISTINCT RECORDS. 

mysql> select student.name,student.section,student_marks.total_marks
        -> from student left join student_marks on
        -> student.rollno=student_marks.rollno
        -> union
        -> select student.name,student.section,student_marks.total_marks
        -> from student right join student_marks on
        -> student.rollno=student_marks.rollno;
