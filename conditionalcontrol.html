THE STRUCTURE OF THE PL/SQL IF STATEMENT IS SIMILAR TO THE STRUCTURE OF IF STATEMENTS IN OTHER PROCEDURAL LANGUAGES.

IF-THEN-END IF

declare
        x number;
        y number;
        begin
        x:=30;
        y:=20;
        if x>y then
            dbms_output.put_line('x is biggest than y');
        end if;
        end;

        
        Statement processed.
        x is biggest than y

IF-THEN-ELSE-END IF

declare
        x number;
        y number;
        begin
        x:=10;
        y:=20;
        if x>y then
            dbms_output.put_line('x is biggest than y');
        else
            dbms_output.put_line('y is biggest than x');
        end if;
        end;
    
        Statement processed.
        y is biggest than x

IF-THEN-ELSIF-END IF

declare
        x number;
        y number;
        z number;
        begin
        x:=10;
        y:=20;
        z:=30;
        if x>y then
            if x>z then
                dbms_output.put_line('x is biggest number');
           else
               dbms_output.put_line('z is biggest number');
            end if;
        elsif y>z then
            dbms_output.put_line('y is biggest number');
        else
            dbms_output.put_line('z is biggest number');
        end if;
        end;
    
        Statement processed.
        z is biggest number
