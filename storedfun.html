A STORED FUNCTION IS A DATABASE OBJECT WHICH CAN BE CALLED MULTIPLE TIMES BECAUSE IT IS STORED IN THE DATABASE PERMANENTLY. THE FUNCTION WILL ALWAYS RETURN A VALUE.

create or replace function fun1(a number,b number) return number is  
        begin 
        return a+b; 
        end; 
         

        Function created.
    
        begin 
        dbms_output.put_line(fun1(10,20)); 
        end; 
         

        Statement processed.
        30
