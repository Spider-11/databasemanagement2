BY DEFAULT, A TABLE COLUMN CAN HOLD NULL VALUES. THE NOT NULL CONSTRAINT ENFORCES A COLUMN TO NOT ACCEPT NULL VALUES.

mysql> create table student(rollno int not null, name varchar(20), age int);
