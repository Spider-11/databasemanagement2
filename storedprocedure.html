A STORED PROCEDURE IS A DATABASE OBJECT WHICH CAN BE CALLED MULTIPLE TIMES BECAUSE IT IS STORED IN THE DATABASE PERMANENTLY.

create or replace procedure PROCEDURE_1(a number,b out number)   
        is  
        begin  
            b:=5+a;
            dbms_output.put_line(a||' '||b);  
        end;
    

        declare 
        c number; 
        begin 
        procedure_1(10,c); 
        dbms_output.put_line(c); 
        end; 
    

        Statement processed.
        10 15
        15
