THE DEFAULT CONSTRAINT IS USED TO INSERT A DEFAULT VALUE INTO A COLUMN.

mysql> create table student4(rollno int, name varchar(20),
        -> city varchar(20) default 'Chandigarh');
        Query OK, 0 rows affected (0.56 sec)
