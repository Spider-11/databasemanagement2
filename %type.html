TO AVOID TYPE AND SIZE CONFLICK BETWEEN A VARIABLE AND THE COLUMN OF A TABLE, THE ATTRIBUTE %TYPE IS USED.

declare
        x student.roll%type;
        y student.name%type;
        begin
        select roll,name into x,y from student where roll=1;
        dbms_output.put_line(x ||' '|| y);
        end;
